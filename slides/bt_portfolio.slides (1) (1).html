Vamos programar uma estratégia que é amplamente discutida na literatura financeira - Value.

O racional por trás da estratégia é comprar ações que estejam baratas em comparação com o universo observado.

Para fazer isto, a estratégia monta um portfólio com as ações que, comparativamente ao universo, estão com o múltiplo de 'price-to-earnings' baixo.

# allocation algorithm names (equally weighted)
class Backtest:
    def __init__(self, returns, ratios, calendar, strategy):
        """Main class to perform the computations.
        Parameters
        ----------
        returns: pandas.DataFrame
            mtrix of returns. 
            must be a valid number.
        calendar: pandas.DatetimeIndex
            The vector of dates to simulate.
            The simulation starts one year after.
        strategy: dict with strategy parameters
            'portfolio_size' (desired size of the portfolio)    
            'distribution' (weight distribution algorithm)
        """

        self.strategy = strategy
        self.dates = returns.index
        self.returns = returns.values
        self.ratios = ratios.values 

        self.calendar = calendar
        self.weights = None
        self._aux_cum_return = None
        self.selected = None

    @property
    def cumulative_return(self):
        x = pd.DataFrame(data=self._aux_cum_return,
                         index=self.dates,
                         columns=['Retorno Acumulado da Estratégia'])
        x = x.replace(0, np.nan).dropna()
        return x

    @staticmethod
    def evolve_weights(returns, weights):
        num = weights * (1 + returns)
        return np.nan_to_num(num / np.sum(num))

    def run(self, verbose=False, is_loading_bar=True):

        if self.weights is not None or self._aux_cum_return:
            raise ValueError("O backtest já foi executado.")

        dates = self.dates

        # inicialização
        self.weights = np.zeros((len(dates), self.returns.shape[1]))
        self._aux_cum_return = np.zeros((len(dates), 1))

        # iteração dos dias
        for i, day in tqdm.notebook.tqdm(enumerate(dates) , total=len(dates), disable=(not is_loading_bar)):

            today = i  # índice atual
            yty = today - 1  # índice do dia anterior

            current_returns = self.returns[today]  # retornos do dia atual

            if today > 0:

                # evolui os pesos do portfolio
                self.weights[today, :] = \
                    self.evolve_weights(current_returns, self.weights[yty, :])

            # se a data é para ser considerada, rodamos o algoritmo
            if day in self.calendar:

                if verbose:
                    print(str(day)[:-9])

                self.stock_picking(yty) #Só olha os dados até o dia anterior para evitar look-ahead-bias
                self.asset_allocation(today)


            # atualiza o retorno da estratégia
            if day > dates[0]:
                # calcula o retorno do portfólio
                rent = np.sum(self.weights[yty] * current_returns)

                # acumula o retorno ao portfólio
                self._aux_cum_return[today] = \
                    (1 + rent) * (1 + self._aux_cum_return[yty]) - 1
    
    def stock_picking(self, day):

        # seleciona o último dado disponível de
        current_ratios = self.ratios[day,]
        
        cut_percentile = np.percentile(current_ratios, (self.strategy['portfolio_size'] / len(current_ratios)) * 100)
        
        self.selected = current_ratios < cut_percentile
        

    def asset_allocation(self, today):

        # igualmente ponderado
        if self.strategy['distribution'] == 'EW':
            self.weights[today] = 0
            self.weights[today, self.selected] = 1 / self.strategy['portfolio_size']
        else:
            raise ValueError("Distribution method not yet implemented.")

# Uma sugestão natural é comparar o gráfico com os retorno acumulado do Ibov no mesmo período

def get_ibov_benchmark_df(backtest_cum_return):
    ibov_rets_df = ibov_df / ibov_df.shift(1) - 1
    ibov_rets_df = ibov_rets_df.loc[backtest_cum_return.index[0]:backtest_cum_return.index[-1]]
    ibov_rets_df = (ibov_rets_df+1).cumprod()-1
    ibov_rets_df.columns=['Índice Ibovespa']
    cum_rets_df = backtest_cum_return.join(ibov_rets_df)
    return cum_rets_df
